#include <stdio.h>
#include <string.h>
#include <ctype.h>

int n, m = 0, p, i = 0, j = 0;
char a[10][10], f[10];

void follow(char c);
void first(char c);

int main() {
    int i, z;
    char c, ch;
    
    printf("Programmed by Vijay Aravindh S. (23BAI1160)\n");
    printf("Enter the number of productions: ");
    scanf("%d", &n);
    
    printf("Enter the productions (e.g., S=Ab):\n");
    for (i = 0; i < n; i++) {
        scanf("%s%c", a[i], &ch);
    }

    do {
        m = 0;
        printf("Enter the non-terminal to find Follow: ");
        scanf(" %c", &c);
        
        follow(c);
        
        printf("Follow(%c) = { ", c);
        for (i = 0; i < m; i++) {
            printf("%c ", f[i]);
        }
        printf("}\n");

        printf("Press 1 to continue, 0 to exit: ");
        scanf("%d", &z);
    } while (z == 1);

    return 0;
}

void follow(char c) {
    // Rule 1: If c is the start symbol, add '$' to Follow(c)
    if (a[0][0] == c) {
        f[m++] = '$';
    }

    for (i = 0; i < n; i++) {
        for (j = 2; j < strlen(a[i]); j++) {
            if (a[i][j] == c) {
                // Case 1: Non-terminal followed by a terminal (e.g., A -> aBb)
                if (a[i][j + 1] != '\0' && !isupper(a[i][j + 1])) {
                    f[m++] = a[i][j + 1];
                }
                // Case 2: Non-terminal followed by another non-terminal (e.g., A -> aBC)
                // This adds First(next_non_terminal) to Follow(c)
                if (a[i][j + 1] != '\0' && isupper(a[i][j + 1])) {
                    first(a[i][j + 1]);
                }
                // Case 3: Non-terminal is at the end of the production (e.g., A -> aB)
                // This adds Follow(lhs) to Follow(c)
                if (a[i][j + 1] == '\0' && c != a[i][0]) {
                    follow(a[i][0]);
                }
            }
        }
    }
}

void first(char c) {
    int k;
    // If the next symbol is a terminal, add it to the set
    if (!isupper(c)) {
        f[m++] = c;
    }
    // Search productions to find the First set of the non-terminal
    for (k = 0; k < n; k++) {
        if (a[k][0] == c) {
            if (a[k][2] == '$') { // Handling epsilon (if applicable)
                return;
            } else if (islower(a[k][2])) {
                f[m++] = a[k][2];
            } else {
                first(a[k][2]);
            }
        }
    }
}
